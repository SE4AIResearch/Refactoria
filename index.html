About Refactoria

Refactoria is a video game that aims to teach high school and university-level computer science students the concepts and skills needed to properly refactor code.

Players play as an expert chef with their sidekick Watson the Whiskbot to refactor Watson's instructions into efficient, readable, and easily maintainable code. While making fun recipes like a fried egg burger, tomato soup, and chicken nuggets, players learn better coding etiquette!

In each of the 13 levels, players are presented with a code-based puzzle that, although it passes the test cases, contains a variety of code smells. They are then tasked with using refactoring techniques to ensure the code passes one or two other requirements designed to teach better coding practices.

Build a Burger


func complete(take_from, put) {
  ACTIVATE("A")
  TAKE("A")
  GOTO("Station 4")
  PLACE("A", put)
}
  
func prep_bread(end_slot) {
  GOTO("Pantry")
  GET("Loaf of Bread")
  GOTO("Station 1")
  PLACE("A", 0)
  complete(1, end_slot)
}
  
func fill_part(ingredient, slot) {
  GOTO("Pantry")
  GET(ingredient)
  GOTO("Station 2")
  PLACE("A", slot)
}
  
func BBQ() {
  fill_part("Tomato", 0)
  fill_part("Sugar", 1)
  fill_part("Vinegar", 2)
  complete(3, 1)
}

Parameter Overload


func prep_ing(ing, mod, ing_slot) {
  GOTO("Pantry")
  GET(ing)
  GOTO("Station 1")
  PLACE(mod, 0)
  ACTIVATE(mod)
  TAKE(mod)
  PLACE("C", ing_slot) }

func end_protocol() {
  ACTIVATE("C")
  TAKE("C")
  GOTO("Window")
  DELIVER()
}

prep_ing("Loaf of Bread", "A", 1)
prep_ing("Raw Chicken", "B", 0)

end_protocol()

An Industrial Kitchen

This solution moves Station 1C to 3A and Station 2B to 3B.


func pre_proc(ing, sta, mod) {
  GOTO("Pantry")
  GET(ing)
  GOTO(sta)
  PLACE(mod, 0)
  ACTIVATE(mod)
  TAKE(mod)
}

func tenders() {
  pre_proc("Loaf of Bread", "Station 1", "A")
  GOTO("Station 2")
  PLACE("C", 1)

func take_order(order) {
  if (order == "Chicken Tenders") {
    tenders()
    GOTO("Window")
    DELIVER()

  } else if (order == "Fried Egg Burger") {
    pre_proc("Loaf of Bread", "Station 1", "A")
    GOTO("Station 2")
    PLACE("A", 0)
    pre_proc("Loaf of Bread", "Station 1", "A")
    GOTO("Station 2")
    PLACE("A", 3)

    fried_egg()
    ACTIVATE("A")
    TAKE("A")

    pre_proc("Loaf of Bread", "Station 1", "A")
    GOTO("Station 2")

    GOTO("Station 2")
    PLACE("A", 3)
    burg_meat()
    bbq_sauce()
    ACTIVATE("A")
    TAKE("A")
    GOTO("Window")
    DELIVER()
  }
}
